  /FORMAT=AVALUE TABLES 
  /STATISTICS=CHISQ CORR 
  /CELLS=COUNT EXPECTED PROP

USE ALL. 
COMPUTE filter_$=(gender = "M"). 
VARIABLE LABELS filter_$ 'gender = "M" (FILTER)'. 
VALUE LABELS filter_$ 0 'Not Selected' 1 'Selected'. 
FORMATS filter_$ (f1.0). 
FILTER BY filter_$. 
EXECUTE.
